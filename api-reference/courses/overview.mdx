# Courses

The Courses API allows you to manage courses in your Konstantly site. You can list courses, get course details, manage assignments, and access course statistics.

## Course Object

<ResponseField name="Course" type="object">
    <Expandable title="Properties">
        <ResponseField name="id" type="integer">
            Unique identifier for the course
        </ResponseField>
        <ResponseField name="name" type="string">
            Name of the course
        </ResponseField>
        <ResponseField name="annotation" type="string">
            Course annotation/description
        </ResponseField>
        <ResponseField name="isDraft" type="boolean">
            Whether the course is in draft state
        </ResponseField>
        <ResponseField name="createdAt" type="integer">
            Timestamp of course creation
        </ResponseField>
        <ResponseField name="updatedAt" type="integer">
            Timestamp of last update
        </ResponseField>
        <ResponseField name="publishedAt" type="integer">
            Timestamp when the course was published
        </ResponseField>
        <ResponseField name="image" type="object" optional>
            Course cover image details
            <Expandable title="Image properties">
                <ResponseField name="id" type="integer">
                    Image identifier
                </ResponseField>
                <ResponseField name="original" type="string">
                    URL to full-size image
                </ResponseField>
                <ResponseField name="mini" type="string">
                    URL to 36x36px thumbnail
                </ResponseField>
                <ResponseField name="small" type="string">
                    URL to 330x130px image
                </ResponseField>
            </Expandable>
        </ResponseField>
    </Expandable>
</ResponseField>
