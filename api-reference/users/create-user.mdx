---
title: 'Create User'
api: 'POST /users'
description: 'Create a new user in your Konstantly site'
---

Create a new user account with specified details, role, and optional course assignments.

## Request

<ParamField header="X-API-KEY" required>
    Your API key for authentication
</ParamField>

### Request Body

<ParamField body="name" type="string" required>
    Full name of the user
</ParamField>

<ParamField body="email" type="string" required>
    User's email address (must be unique)
</ParamField>

<ParamField body="password" type="string" required>
    User's password
</ParamField>

<ParamField body="roleAlias" type="string" required>
    Role alias to assign to the user
</ParamField>

<ParamField body="id" type="string">
    Custom user ID (optional, generated automatically if not provided)
</ParamField>

<ParamField body="bio" type="string">
    User's bio description
</ParamField>

<ParamField body="occupation" type="string">
    User's job title
</ParamField>

<ParamField body="location" type="string">
    User's location
</ParamField>

<ParamField body="timezone" type="string">
    User's timezone in TZ format (e.g., "Europe/London")
</ParamField>

<ParamField body="fromApi" type="boolean">
    Whether the user data is editable via web interface (defaults to true)
</ParamField>

<ParamField body="courses" type="array">
    Array of course IDs to assign to the user
</ParamField>

<ParamField body="deadlineAt" type="integer">
    Timestamp for course assignment deadlines
</ParamField>

<ParamField body="groups" type="array">
    Array of group IDs to add the user to
</ParamField>

<ParamField body="attributes" type="object">
    Custom user attributes
</ParamField>
